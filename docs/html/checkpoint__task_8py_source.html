79 def _heading_change(self, current, start):
80 diff = current - start
81 while diff > 180.0:
82 diff -= 360.0
83 while diff < -180.0:
84 diff += 360.0
85 return diff

87 def _save_pos(self):
88 self._x0 = self._est_x.get()
89 self._y0 = self._est_y.get()
90
91 def _dist_sq(self):
92 dx = self._est_x.get() - self._x0
93 dy = self._est_y.get() - self._y0
94 return dx * dx + dy * dy

96 def _spin_steer(self, cw):

120 def run(self):
121 while True:
122
123 # ── S0: Idle ──────────────────────────────────────────────────────
124 if self._state == S0_IDLE:
125 if self._checkpoint_go.get():
126 self._start_heading = self._imu.get_heading()
127 self._hdg_ctr = 0
128
129 # Use 500 s motor timeout for the full course
130 self._time_disable.put(True)
131
132 self._setpoint.put(float(FAST_SPEED_MM_S))
133 self._steer.put(0.0)
134 self._line_enable.put(False) # no line follow during sprint
135 self._leftMotorGo.put(True)
136 self._rightMotorGo.put(True)
137 self._obs_go.put(True)
138
139 print("CP: sprint start, hdg ref={:.1f} deg.".format(
140 self._start_heading))
141 self._state = S1_FAST
142
143 # ── S1: High-speed sprint ─────────────────────────────────────────
144 elif self._state == S1_FAST:
145 if not self._leftMotorGo.get():
146 self._motor_timeout("sprint")
147 else:
148 # Heading hold: correct drift proportionally using IMU
149 self._hdg_ctr += 1
150 if self._hdg_ctr >= 3:
151 self._hdg_ctr = 0
152 drift = self._heading_change(
153 self._imu.heading, self._start_heading)
154 corr = drift * SPRINT_HDG_GAIN
155 if corr > SPRINT_STEER_LIMIT:
156 corr = SPRINT_STEER_LIMIT
157 elif corr < -SPRINT_STEER_LIMIT:
158 corr = -SPRINT_STEER_LIMIT
159 self._steer.put(corr)
160 if self._leftMotorGo.get() and self._est_x.get() >= SLOWDOWN_DIST_MM:
161 self._setpoint.put(float(SLOW_SPEED_MM_S))
162 self._hdg_ctr = 0
163 self._pre_spinning = False
164 self._line_enable.put(True) # enable line follow for the curve
165 print("CP: {:.0f} mm — slowing to {} mm/s for turn.".format(
166 self._est_x.get(), SLOW_SPEED_MM_S))
167 self._state = S2_APPROACH
168
169 # ── S2: Slow approach, detect 87° heading change ──────────────────
170 elif self._state == S2_APPROACH:
171 if not self._leftMotorGo.get():
172 self._motor_timeout("approach")
173 else:
174 if self._pre_spinning:
175 # Keep applying spin steer every tick while finishing the turn
176 self._steer.put(self._spin_steer(self._cw))
177 self._hdg_ctr += 1
178 if self._hdg_ctr >= 5:
179 self._hdg_ctr = 0
180 signed = self._heading_change(
181 self._imu.heading, self._start_heading)
182 if abs(signed) >= HEADING_1_DEG:
183 self._cw = (signed < 0) # record turn direction
184 self._line_enable.put(False)
185 self._steer.put(0.0)
186 self._setpoint.put(float(SLOW_SPEED_MM_S))
187 self._save_pos()
188 print("CP: turn 1 ({:.1f} deg, cw={}) — fwd {} mm.".format(
189 signed, self._cw, STRAIGHT_2_MM))
190 self._state = S3_STRAIGHT
191 elif abs(signed) >= HEADING_PRE_SPIN_DEG and not self._pre_spinning:
192 # Line has probably ended; take over with IMU-guided spin
193 self._cw = (signed < 0)
194 self._pre_spinning = True
195 self._line_enable.put(False)
196 self._steer.put(self._spin_steer(self._cw))
197 self._setpoint.put(0.0)
198 print("CP: pre-spin at {:.1f} deg — finishing turn IMU-guided.".format(
199 signed))
200
201 # ── S3: Forward 150 mm (no line follow) ───────────────────────────
202 elif self._state == S3_STRAIGHT:
203 if not self._leftMotorGo.get():
204 self._motor_timeout("straight 2")
205 elif self._dist_sq() >= STRAIGHT_2_MM * STRAIGHT_2_MM:
206 self._setpoint.put(0.0)
207 self._steer.put(self._spin_steer(self._cw))
208 self._ref_heading = self._imu.heading
209 self._hdg_ctr = 0
210 print("CP: {} mm done — spin 2 ({}).".format(
211 STRAIGHT_2_MM, "CW" if self._cw else "CCW"))
212 self._state = S4_SPIN_2
213
214 # ── S4: Pivot to face wall (start_heading + 180°) ─────────────────
215 elif self._state == S4_SPIN_2:
216 if not self._leftMotorGo.get():
217 self._motor_timeout("spin 2")
218 else:
219 # Absolute heading target: always stop at start+180° regardless
220 # of how many degrees turn 1 consumed. Checked every tick to
221 # minimise overshoot (same pattern as S11_SPIN_5).
222 target = self._start_heading + 180.0
223 err = self._heading_change(target, self._imu.heading)
224 if abs(err) < 5.0:
225 self._steer.put(0.0)
226 self._setpoint.put(float(WALL_SPEED_MM_S))
227 self._save_pos()
228 print("CP: spin 2 done (err={:.1f} deg) — heading to wall.".format(
229 err))
230 self._state = S5_WALL
231 else:
232 spin_cw = (err < 0)
233 self._steer.put(self._spin_steer(spin_cw))
234
235 # ── S5: Drive to wall, wait for bump ──────────────────────────────
236 elif self._state == S5_WALL:
237 if not self._leftMotorGo.get():
238 self._motor_timeout("wall run")
239 elif self._bump.value(): # pull-down: 1 = bumped
240 self._setpoint.put(-float(REVERSE_SPEED_MM_S))
241 self._steer.put(0.0)
242 self._save_pos()
243 print("CP: wall hit — reversing {} mm.".format(REVERSE_DIST_MM))
244 self._state = S6_REVERSE
245
246 # ── S6: Reverse REVERSE_DIST_MM ───────────────────────────────────
247 elif self._state == S6_REVERSE:
248 if not self._leftMotorGo.get():
249 self._motor_timeout("reverse")
250 elif self._dist_sq() >= REVERSE_DIST_MM * REVERSE_DIST_MM:
251 self._setpoint.put(0.0)
252 self._steer.put(self._spin_steer(not self._cw))
253 self._ref_heading = self._imu.heading
254 self._hdg_ctr = 0
255 print("CP: reversed {} mm — spin 3 ({}).".format(
256 REVERSE_DIST_MM, "CCW" if self._cw else "CW"))
257 self._state = S7_SPIN_3
258
259 # ── S7: Pivot 90° opposite direction ──────────────────────────────
260 elif self._state == S7_SPIN_3:
261 if not self._leftMotorGo.get():
262 self._motor_timeout("spin 3")
263 else:
264 self._steer.put(self._spin_steer(not self._cw))
265 self._hdg_ctr += 1
266 if self._hdg_ctr >= 5:
267 self._hdg_ctr = 0
268 change = abs(self._heading_change(
269 self._imu.heading, self._ref_heading))
270 if change >= HEADING_SPIN_DEG:
271 self._steer.put(0.0)
272 self._setpoint.put(float(SLOW_SPEED_MM_S))
273 self._save_pos()
274 print("CP: spin 3 done ({:.1f} deg) — fwd {} mm.".format(
275 change, STRAIGHT_3_MM))
276 self._state = S8_STRAIGHT_3
277
278 # ── S8: Forward STRAIGHT_3_MM (no line follow) ────────────────────
279 elif self._state == S8_STRAIGHT_3:
280 if not self._leftMotorGo.get():
281 self._motor_timeout("straight 3")
282 elif self._dist_sq() >= STRAIGHT_3_MM * STRAIGHT_3_MM:
283 self._setpoint.put(0.0)
284 self._steer.put(self._spin_steer(self._cw))
285 self._ref_heading = self._imu.heading
286 self._hdg_ctr = 0
287 print("CP: {} mm done — spin 4 (CW).".format(STRAIGHT_3_MM))
288 self._state = S9_SPIN_4
289
290 # ── S9: Pivot 90° CW ──────────────────────────────────────────────
291 elif self._state == S9_SPIN_4:
292 if not self._leftMotorGo.get():
293 self._motor_timeout("spin 4")
294 else:
295 self._steer.put(self._spin_steer(self._cw))
296 self._hdg_ctr += 1
297 if self._hdg_ctr >= 5:
298 self._hdg_ctr = 0
299 change = abs(self._heading_change(
300 self._imu.heading, self._ref_heading))
301 if change >= HEADING_SPIN_DEG:
302 self._steer.put(0.0)
303 self._setpoint.put(float(SLOW_SPEED_MM_S))
304 self._line_enable.put(True)
305 self._save_pos()
306 print("CP: spin 4 done ({:.1f} deg) — line follow {} mm.".format(
307 change, SECOND_LINE_DIST_MM))
308 self._state = S10_LINE2
309
310 # ── S10: Line follow until total dist >= SECOND_LINE_DIST_MM ─────
311 elif self._state == S10_LINE2:
312 if not self._leftMotorGo.get():
313 self._motor_timeout("line 2")
314 elif self._dist_sq() >= SECOND_LINE_DIST_MM * SECOND_LINE_DIST_MM:
315 self._line_enable.put(False)
316 self._setpoint.put(0.0)
317 self._steer.put(0.0)
318 self._hdg_ctr = 0
319 print("CP: second line done — spinning to {:.0f} deg from start.".format(
320 SPIN5_TARGET_DEG))
321 self._state = S11_SPIN_5
322
323 # ── S11: Rotate to SPIN5_TARGET_DEG from original start heading ───
324 elif self._state == S11_SPIN_5:
325 if not self._leftMotorGo.get():
326 self._motor_timeout("spin 5")
327 else:
328 # Compute shortest-arc error to target heading
329 target = self._start_heading + SPIN5_TARGET_DEG
330 err = self._heading_change(target, self._imu.heading)
331 if abs(err) < 5.0:
332 self._steer.put(0.0)
333 self._setpoint.put(float(FINAL_SPEED_MM_S))
334 self._save_pos()
335 print("CP: spin 5 done — forward {} mm.".format(STRAIGHT_4_MM))
336 self._state = S12_FORWARD_4
337 else:
338 # Spin toward target: err>0 means target is CCW, err<0 means CW
339 spin_cw = (err < 0)
340 self._steer.put(self._spin_steer(spin_cw))
341
342 # ── S12: Drive forward STRAIGHT_4_MM ──────────────────────────────
343 elif self._state == S12_FORWARD_4:
344 if not self._leftMotorGo.get():
345 self._motor_timeout("forward 4")
346 elif self._dist_sq() >= STRAIGHT_4_MM * STRAIGHT_4_MM:
347 self._setpoint.put(0.0)
348 self._ref_heading = self._imu.heading
349 print("CP: {} mm done — spin 6 ({:.0f} deg).".format(
350 STRAIGHT_4_MM, SPIN6_DEG))
351 self._state = S13_SPIN_6
352
353 # ── S13: Rotate SPIN6_DEG from current heading ────────────────────
354 elif self._state == S13_SPIN_6:
355 if not self._leftMotorGo.get():
356 self._motor_timeout("spin 6")
357 else:
358 target = self._ref_heading + SPIN6_DEG
359 err = self._heading_change(target, self._imu.heading)
360 if abs(err) < 5.0:
361 self._steer.put(0.0)
362 self._setpoint.put(float(FINAL_SPEED_MM_S))
363 self._save_pos()
364 print("CP: spin 6 done — forward {} mm.".format(STRAIGHT_5_MM))
365 self._state = S14_FORWARD_5
366 else:
367 spin_cw = (err < 0)
368 self._steer.put(self._spin_steer(spin_cw))
369
370 # ── S14: Drive forward STRAIGHT_5_MM ──────────────────────────────
371 elif self._state == S14_FORWARD_5:
372 if not self._leftMotorGo.get():
373 self._motor_timeout("forward 5")
374 elif self._dist_sq() >= STRAIGHT_5_MM * STRAIGHT_5_MM:
375 print("CP: {} mm done — full stop.".format(STRAIGHT_5_MM))
376 self._state = S15_STOP
377
378 # ── S15: Full stop ─────────────────────────────────────────────────
379 elif self._state == S15_STOP:
380 self._stop_all()
381 print("CP: sequence complete.")
382 self._state = S0_IDLE
383
384 yield self._state
