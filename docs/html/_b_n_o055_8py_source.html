161 def get_calibration_status(self):
162 """
163 Returns calibration status as a dict (sys, gyro, accel, mag).
164 Each value is 0 (uncalibrated) to 3 (fully calibrated).
165 """
166 byte = self.read8(self.addr, self.REG_CALIB_STAT)
167 return {
168 "sys": (byte >> 6) & 0x03,
169 "gyro": (byte >> 4) & 0x03,
170 "accel": (byte >> 2) & 0x03,
171 "mag": (byte >> 0) & 0x03,
172 }

174 def is_fully_calibrated(self):
175 """Returns True when gyro, accel and mag are all at level 3."""
176 s = self.get_calibration_status()
177 return s["gyro"] == 3 and s["accel"] == 3 and s["mag"] == 3

188 def set_calibration_coefficients(self, data):
189 """Write 22 bytes of saved calibration coefficients back to the IMU."""
190 if len(data) != 22:

192 prev_mode = self.mode

194 time.sleep_ms(30)

196 self.write8(self.addr, self.REG_CALIB_DATA + i, byte)

201 def _run_calibration(self):

203 # In IMU mode the magnetometer is disabled — don't wait for it
204 imu_mode = self._mode in ("IMU",)

231 def _save_calibration(self):
232 """Read coefficients in CONFIG mode and write to file."""
233 try:
234 self.set_mode("CONFIG")
235 time.sleep_ms(100)
236 coeffs = bytes(self.readn(self.addr, self.REG_CALIB_DATA, 22))
237 self.set_mode(self._mode)
