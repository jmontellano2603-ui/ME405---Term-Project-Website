12# ── Discretized observer matrices ─────────────────────────────────────────────
13# Computed using professor's method (LQE + c2d ZOH) with:
14# r=35mm, w=141mm, tau=0.057s, Km=9.26/35 rad/s per effort, Ts=0.031s
15# Q=diag([1,1,10,10]), R=diag([0.1,0.1,0.01,1.0])
16#
17# State vector: x = [s (mm), psi (rad), omL (rad/s), omR (rad/s)]
18# Input vector: u* = [uL, uR, sL (mm), sR (mm), psi (rad), psidot (rad/s)]
19
20AD = np.array([
21 [ 0.82531337, 0.00000000, 0.38062706, 0.38062706],
22 [ 0.00000000, 0.00005588, -0.00048319, 0.00048319],
23 [-0.07731724, 0.00045768, 0.56043394, -0.01943210],
24 [-0.07731724, -0.00045768, -0.01943210, 0.56043394]
25])
26
27BD = np.array([
28 [ 0.02426532, 0.02426532, 0.08734331, 0.08734331, 0.00000000, 0.00000000],
29 [-0.00006277, 0.00006277, -0.00708468, 0.00708468, 0.00100492, 0.00000245],
30 [ 0.08603749, -0.00080431, 0.03866186, 0.03865538, -0.00000046, -0.00168906],
31 [-0.00080431, 0.08603749, 0.03865538, 0.03866186, 0.00000046, 0.00168906]
32])

34# C is exact from kinematics — state uses angular wheel speeds (rad/s)
35# Row 3: psidot = (-r/w)*omL + (r/w)*omR where r/w = 35/141 = 0.24823
36C_mat = np.array([
37 [1.0, -70, 0.0, 0.0 ],
38 [1.0, 70, 0.0, 0.0 ],
39 [0.0, 1.0, 0.0, 0.0 ],
40 [0.0, 0.0, -0.24822695, 0.24822695]
41])

43_MM_PER_COUNT = 0.153 # encoder counts to mm
44_DEG_TO_RAD = math.pi / 180.0
45_R = 35.0 # wheel radius [mm] — needed to get v from omega

48class task_state:
49 """
50 Discrete-time Luenberger state observer for the Romi robot.
51
52 State vector: x = [s (mm), psi (rad), omL (rad/s), omR (rad/s)]
53 Update law: x_{k+1} = AD @ x_k + BD @ u*_k
54 where u* = [uL, uR, sL, sR, psi, psidot]
55 """

57 def __init__(self, imu, enc_left, enc_right,
58 effort_left: Share,
59 effort_right: Share,
60 go_flag: Share,
61 est_x: Share,
62 est_y: Share,
63 est_heading: Share,
64 est_vel: Share):

66 self._imu = imu

72 self._est_x = est_x
73 self._est_y = est_y
74 self._est_hdg = est_heading
75 self._est_vel = est_vel

77 # State as plain floats between ticks
78 self._s = 0.0 # forward displacement (mm)
79 self._psi = 0.0 # heading (rad)
80 self._omL = 0.0 # left wheel angular speed (rad/s)
81 self._omR = 0.0 # right wheel angular speed (rad/s)
82
83 self._pos_x = 0.0
84 self._pos_y = 0.0
85 self._print_ctr = 0
86 self._state = S0_INIT

91 def _encoder_mm(self, enc):
92 return enc.get_position() * _MM_PER_COUNT

94 def _reset(self):
95 self._s = 0.0
96 self._psi = 0.0
97 self._omL = 0.0
98 self._omR = 0.0
99 self._pos_x = 0.0
100 self._pos_y = 0.0
101 self._last_t = ticks_us()
102 self._enc_l.zero()
103 self._enc_r.zero()

105 def run(self):
106 while True:
107
108 if self._state == S0_INIT:
109 self._reset()
110 self._state = S1_WAIT
111
112 elif self._state == S1_WAIT:
113 if self._go.get():
114 self._reset()
115 print("State observer: starting.")
116 self._state = S2_RUN
117
118 elif self._state == S2_RUN:
119 if not self._go.get():
120 print("State observer: stopped.")
121 self._state = S1_WAIT
122
123 else:
124 # -- 1. Read sensors --------------------------------------
125 sL = self._encoder_mm(self._enc_l)
126 sR = self._encoder_mm(self._enc_r)
127 self._imu.update()
128 psi = self._imu.heading * _DEG_TO_RAD # rad
129 psi_dot = self._imu.yaw_rate * _DEG_TO_RAD # rad/s
130 uL = self._eff_l.get()
131 uR = self._eff_r.get()
132
133 # -- 2. Build state and input vectors ---------------------
134 x = np.array([[self._s ],
135 [self._psi],
136 [self._omL],
137 [self._omR]])
138
139 u_star = np.array([[uL ],
140 [uR ],
141 [sL ],
142 [sR ],
143 [psi ],
144 [psi_dot]])
145
146 # -- 3. Observer update -----------------------------------
147 x_new = np.dot(AD, x) + np.dot(BD, u_star)
148
149 # -- 4. Unpack to plain floats ----------------------------
150 self._s = x_new[0, 0]
151 self._psi = x_new[1, 0]
152 self._omL = x_new[2, 0]
153 self._omR = x_new[3, 0]
154
155 # -- 5. Forward velocity ----------------------------------
156 # omL/omR are in rad/s so multiply by r to get mm/s
157 v_hat = _R * (self._omL + self._omR) / 2.0
158
159 # -- 6. Integrate x/y position ----------------------------
160 now = ticks_us()
161 dt = ticks_diff(now, self._last_t) / 1_000_000.0
162 self._last_t = now
163 self._pos_x += v_hat * math.cos(self._psi) * dt
164 self._pos_y += v_hat * math.sin(self._psi) * dt
165
166 # -- 7. Compute y_hat = C @ x -----------------------------
167 y_hat = np.dot(C_mat, x_new)
168
169 sL_hat = y_hat[0, 0]
170 sR_hat = y_hat[1, 0]
171 psi_hat_deg = y_hat[2, 0] / _DEG_TO_RAD
172 psidot_hat = y_hat[3, 0] / _DEG_TO_RAD # deg/s
173
174 # -- 8. Publish to shares ---------------------------------
175 self._est_x.put(self._pos_x)
176 self._est_y.put(self._pos_y)
177 self._est_hdg.put(psi_hat_deg)
178 self._est_vel.put(v_hat)
179
180 # -- 9. Print ~once per second ----------------------------
181 self._print_ctr += 1
182 if self._print_ctr >= 32:
183 self._print_ctr = 0
184 print(
185 "y_hat: sL={:.1f} sR={:.1f} psi={:.1f}deg "
186 "psidot={:.1f}deg/s | x={:.1f}mm y={:.1f}mm".format(
187 sL_hat, sR_hat, psi_hat_deg, psidot_hat,
188 self._pos_x, self._pos_y))
189
190 yield self._state
